library(osmdata)

library(tidyverse)

Load locations

cambridge_polling <- st_read(
  "data-downloads/ELECTIONS_PollingLocations/ELECTIONS_PollingLocations.shp")

cambridge_polling_lat_lng <- cambridge_polling %>%
  st_transform(crs = 4326)

Get OpenStreetMap data for use by OpenTripPlanner

Create a folder in your Repo called OTP. Within that folder, create another folder called graphs. Within that folder, create another folder called default.

cambridge_street_query <- opq(bbox = 'Cambridge MA USA') %>%
  add_osm_feature(key = 'highway')

cambridge_street_data <- cambridge_street_query %>%
  osmdata_xml(file = 'OTP/graphs/default/cambridge_streets.osm')

cambridge_street_features <- cambridge_street_query %>%
  osmdata_sf()

cambridge_streets <- cambridge_street_features$osm_lines %>%
  st_transform(crs = st_crs(cambridge_polling))

Verify that you now have a file in your default directory called cambridge_streets.osm, and try plotting the streets youve downloaded

ggplot(cambridge_streets) +
  geom_sf() +
  theme_map()

Now we need to download a little java utility called otp.jar and save it to the OPT directory weve created

path_otp <- otp_dl_jar("OTP")
## The OTP will be saved to OTP/otp.jar

Now, were ready to start the OpenTripPlanner application!

path_data <- file.path(getwd(), "OTP")
path_otp <- paste(path_data,"otp.jar",sep = "/")

otp_build_graph(otp = path_otp, dir = path_data, memory = 1024)

After you do this next part, an OTP application will open in you web browser. This indicates that OpenTripPlanner is running, but other than that, you can ignore it.

otp_setup(otp = path_otp, dir = path_data, memory = 1024)
## 2020-09-28 14:41:57 OTP is loading and may take a while to be useable

# Connect to opentripplanner
otpcon <- otp_connect()

Create Isochrone for area within a ten-minute walk.

iso_10min_walk <- 
  otp_isochrone(otpcon = otpcon, fromPlace = cambridge_polling_lat_lng, 
                mode = "walk", cutoffSec = 600) %>%
  st_transform(crs = st_crs(cambridge_polling))

Plot the isochrone.

right_side <- st_bbox(iso_10min_walk)$xmax
left_side  <- st_bbox(iso_10min_walk)$xmin
top_side <- st_bbox(iso_10min_walk)$ymax
bottom_side <- st_bbox(iso_10min_walk)$ymin

ggplot(iso_10min_walk) +
  geom_sf(fill = "pink") +
  geom_sf(data = cambridge_streets) +
  geom_sf(data = cambridge_polling) +
    coord_sf(xlim = c(left_side, right_side), 
           ylim = c(bottom_side, top_side), expand = FALSE)
